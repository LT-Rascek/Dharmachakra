dharma_hinduism.005 = {# Zealous Child Seeks to become a Monk
	type = character_event
	title = dharma_hinduism.005.t
	desc = dharma_hinduism.005.desc
	theme = faith
	override_background = {
		event_background = throne_room
	}
	left_portrait = {
		character = root
		animation = personality_rational
	}	

	trigger = {
		faith = { has_doctrine = tenet_monasticism }
		is_at_war = no
		religion = religion:hinduism_religion
		is_landed = yes	
		random_child = {# find a random Zealous Child and have him ask to be a monk
			limit = {
				age >= 20
				has_trait = zealous
				NOT = { has_trait = devoted }
			}
			save_scope_as = child_wants_monk
		}
	}

	option = {# You can be come a monk
		name = dharma_hinduism.005.blessings
		scope:child_wants_monk = {
			add_trait = devoted
		}
	}
	option = {# I do not allow it!
		name = dharma_hinduism.005.no
		scope:heir = {
			add_opinion = {
				target = root
				modifier = angry_opinion
				opinion = -20
			}
		}
	}
}

dharma_hinduism.011 = {# Pumsavana - Blessing for the child in the womb
	type = character_event
	title = dharma_hinduism.011.t
	desc = {
		first_valid = {
			triggered_desc = {# Mother
				trigger = { is_female = yes }
				desc = dharma_hinduism.011.desc.pumsavana_mother
			}
			triggered_desc = {# Father
				trigger = { is_male = yes }
				desc = dharma_hinduism.011.desc.pumsavana_father
			}
		}
	}
	theme = faith
	override_background = {
		event_background = throne_room
	}
	left_portrait = {
		character = root
		animation = personality_rational
	}	

	trigger = {
		is_landed = yes
		religion = religion:hinduism_religion
		is_landed = yes
		OR = {
			AND = {# Mother
				is_female = yes
				NOT = { has_character_flag = spouse_pumsavana }
				has_trait = pregnant
				save_scope_as = spouse_pregnant_hindu
			}
			AND = {# Father
				is_male = yes
					any_spouse = {
					NOT = { has_character_flag = spouse_pumsavana }
					has_trait = pregnant
					save_scope_as = spouse_pregnant_hindu
				}
			}
		}
		
	}

	option = {# May the Gods Bless our child
		name = {# Mother
			trigger = { is_female = yes }
			text = dharma_hinduism.011.mother
			add_piety = 10
			stress_impact = {
				base = minor_stress_impact_loss
			}
		}
		name = {# Father
			trigger = { is_male = yes }
			text = dharma_hinduism.011.father
			add_piety = 5
			stress_impact = {
				base = miniscule_stress_impact_loss
			}
		}
	}

	after = {
		hidden_effect = {
			spouse_pregnant_hindu = {
				if = {
					limit = { NOT = { has_character_flag = spouse_pumsavana } }
					add_character_flag = spouse_pumsavana
				}
			}
		}
	}

}
