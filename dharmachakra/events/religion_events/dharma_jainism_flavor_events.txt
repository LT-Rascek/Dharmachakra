dharma_jainism.006 = {# Non-Indian merchants upset about naked Jain ascetics
	type = character_event
	title = dharma_jainism.006.t
	desc = dharma_jainism.006.desc
	theme = faith
	override_background = {
		event_background = throne_room
	}
	left_portrait = {
		character = root
		animation = personality_rational
	}	

	trigger = {
		is_at_war = no
		is_landed = yes	
		capital_province = {
			geographical_region = world_india
		}
	}

	immediate = {
		random_sub_realm_barony = {
			title_province = {
				county = {
					limit = {
						geographical_region = world_india
						is_coastal = yes
					}
					save_scope_as = guild_county
				}
			}
		}
	}

	option = {# So What
		name = dharma_jainism.006.a
		scope:guild_county = {
			add_county_modifier = {
				modifier = stewardship_domain_special_1500_resentful_guilds
				days = 1825
			}
		}
	}
	option = {# tell the Jain ascetics to move
		trigger = { NOT = { religion = religion:jainism_religion } }
		name = dharma_jainism.006.move_monk
		any_courtier = {
			limit = { religion = religion:jainism_religion }
			add_opinion = {
				target = root
				modifier = impious_opinion
			}
		}
		any_vassal = {
			limit = { religion = religion:jainism_religion}
			add_opinion = {
				target = root
				modifier = impious_opinion
			}
		}
	}

}
