dharma_set_up_next_nalanda_event_effect = {# Visiting Nalanda
	scope:activity = {
		change_variable = {
			name = dharma_nalanda_event_counter
			add = 1
		}
	}
	save_scope_value_as = {
		name = event_counter
		value = scope:activity.var:dharma_nalanda_event_counter
	}
	if = {
		limit = { # TODO: Is any fallback handling needed in an else here?
			exists = scope:activity
		}
		if = {
			limit = {
				scope:activity.var:dharma_nalanda_event_counter = {
					compare_value <= 5
				}
			}
			trigger_event = {
				on_action = dharma_nalanda_journey_events
				days = $DAYS$
			}
		}
		else = {
			if = { #For event background 
				limit = { scope:activity.var:dharma_nalanda_visit_done = no }
				scope:activity = {
					set_variable = {
						name = dharma_nalanda_visit_done
						value = yes
					}
				}
			}
			trigger_event = {
				on_action = dharma_nalanda_visit_over_events
				days = $DAYS$
			}
		}
		scope:activity = {
			change_variable = {
				name = dharma_nalanda_event_counter
				add = 1
			}
		}
	}
}

		name = event_counter
		value = scope:activity.var:dharma_buddhist_retreat_event_counter
	}
	if = {
		limit = { # TODO: Is any fallback handling needed in an else here?
			exists = scope:activity
		}
		if = {
			limit = {
				scope:activity.var:dharma_buddhist_retreat_event_counter = {
					compare_value <= scope:activity.var:dharma_buddhist_retreate_length_value 
				}
			}
			trigger_event = {
				on_action = dharma_buddhist_retreat_events
				days = $DAYS$
			}
		}
		else = {
			if = { #For event background 
				limit = { scope:activity.var:dharma_buddhist_retreat_done = no }

					set_variable = {
						name = dharma_buddhist_retreat_done

				}
			}
			trigger_event = {
				on_action = dharma_buddhist_retreat_over_events
				days = $DAYS$
			}
		}
		scope:activity = {
			change_variable = {
				name = dharma_buddhist_retreat_event_counter
				add = 1
			}
		}
	}
